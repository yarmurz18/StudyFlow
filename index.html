using System; using System.Collections.Generic; using System.Linq; class Program { class Player { public int Id { get; set; } public string Name { get; set; } public int Level { get; set; } } class Item { public int Id { get; set; } public string Name { get; set; } public int Power { get; set; } } class Inventory { public int PlayerId { get; set; } public int ItemId { get; set; } public int Quantity { get; set; } } static void Main() { var players = new List { new Player { Id = 1, Name = "Arthas", Level = 10 }, new Player { Id = 2, Name = "Illidan", Level = 15 }, new Player { Id = 3, Name = "Thrall", Level = 8 }, new Player { Id = 4, Name = "Jaina", Level = 12 }, new Player { Id = 5, Name = "Jaina2", Level = 13 } }; var items = new List { new Item { Id = 1, Name = "Sword", Power = 10 }, new Item { Id = 2, Name = "Staff", Power = 8 }, new Item { Id = 3, Name = "Axe", Power = 12 }, new Item { Id = 4, Name = "Bow", Power = 7 } }; var inventory = new List { new Inventory { PlayerId = 1, ItemId = 1, Quantity = 1 }, new Inventory { PlayerId = 1, ItemId = 2, Quantity = 2 }, new Inventory { PlayerId = 2, ItemId = 3, Quantity = 1 }, new Inventory { PlayerId = 3, ItemId = 4, Quantity = 3 } }; Console.WriteLine("=== 1. PlayerName + ItemId ==="); var playerItemIds = players.Join( inventory, p => p.Id, i => i.PlayerId, (p, i) => new { p.Name, i.ItemId } ); foreach (var item in playerItemIds) { Console.WriteLine($"Гравець: {item.Name}, ItemId: {item.ItemId}"); } Console.WriteLine("\n=== 2. ItemName + Quantity ==="); var itemQuantities = inventory.Join( items, inv => inv.ItemId, it => it.Id, (inv, it) => new { it.Name, inv.Quantity } ); foreach (var item in itemQuantities) { Console.WriteLine($"Предмет: {item.Name}, Кількість: {item.Quantity}"); } Console.WriteLine("\n=== 3. Гравці, у яких є предмети (без дублікатів) ==="); var playersWithItems = players .Where(p => inventory.Any(i => i.PlayerId == p.Id)) .Select(p => p.Name); foreach (var playerName in playersWithItems) { Console.WriteLine($"Гравець: {playerName}"); } Console.WriteLine("\n=== ДОМАШНЄ ЗАВДАННЯ: Кількість записів inventory для кожного гравця ==="); var inventoryCountPerPlayer = players .GroupJoin( inventory, p => p.Id, i => i.PlayerId, (p, invItems) => new { PlayerName = p.Name, ItemsCount = invItems.Count() } ) .Where(p => p.ItemsCount > 0); // Тільки гравці з предметами foreach (var player in inventoryCountPerPlayer) { Console.WriteLine($"Гравець: {player.PlayerName}, Кількість записів в інвентарі: {player.ItemsCount}"); } // Альтернативний варіант підрахунку (включаючи гравців з 0 предметів) Console.WriteLine("\n=== Альтернативний варіант (всі гравці) ==="); var allPlayersInventoryCount = from p in players join i in inventory on p.Id equals i.PlayerId into invGroup select new { PlayerName = p.Name, ItemsCount = invGroup.Count() }; foreach (var player in allPlayersInventoryCount) { Console.WriteLine($"Гравець: {player.PlayerName}, Кількість записів: {player.ItemsCount}"); } // Додатково: загальна кількість предметів (сума Quantity) для кожного гравця Console.WriteLine("\n=== БОНУС: Загальна кількість предметів (сума Quantity) для кожного гравця ==="); var totalItemsPerPlayer = from p in players join i in inventory on p.Id equals i.PlayerId into invGroup select new { PlayerName = p.Name, TotalItems = invGroup.Sum(x => x.Quantity) }; foreach (var player in totalItemsPerPlayer.Where(p => p.TotalItems > 0)) { Console.WriteLine($"Гравець: {player.PlayerName}, Всього предметів: {player.TotalItems}"); } } }
